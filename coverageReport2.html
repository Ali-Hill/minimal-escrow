   195    meetsTarget :: TxInfo -> EscrowTarget DatumHash -> Bool
   196    meetsTarget ptx = \case
   197        PaymentPubKeyTarget pkh vl ->
   198            valuePaidTo ptx (unPaymentPubKeyHash pkh) `geq` vl
   199        ScriptTarget validatorHash dataValue vl ->
   200            case scriptOutputsAt validatorHash ptx of
   201                [(dataValue', vl')] ->
   202                    traceIfFalse "dataValue" (dataValue' == (OutputDatumHash dataValue))
   203                    && traceIfFalse "value" (vl' `geq` vl)
   204                _ -> False

   207    validate :: EscrowParams DatumHash -> PaymentPubKeyHash -> Action -> ScriptContext -> Bool
   208    validate EscrowParams{escrowDeadline, escrowTargets} contributor action ScriptContext{scriptContextTxInfo} =
   209        case action of
   210            Redeem ->
   211                traceIfFalse "escrowDeadline-after" (escrowDeadline `after` txInfoValidRange scriptContextTxInfo)
   212                && traceIfFalse "meetsTarget" (all (meetsTarget scriptContextTxInfo) escrowTargets)
   213            Refund ->
   214                traceIfFalse "escrowDeadline-before" ((escrowDeadline - 1) `before` txInfoValidRange scriptContextTxInfo)
   215                && traceIfFalse "txSignedBy" (scriptContextTxInfo `txSignedBy` unPaymentPubKeyHash contributor)
