   231    meetsTarget :: TxInfo -> EscrowTarget Datum -> Bool
   232    meetsTarget ptx = \case
   233      PaymentPubKeyTarget pkh vl ->
   234        valuePaidTo ptx (unPaymentPubKeyHash pkh) `geq` vl
   235      ScriptTarget validatorHash dataValue vl ->
   236        case scriptOutputsAt validatorHash ptx of
   237          [(dataValue', vl')] ->
   238            PlutusTx.traceIfFalse "dataValue" (dataValue' PlutusTx.== OutputDatum dataValue)
   239              PlutusTx.&& PlutusTx.traceIfFalse "value" (vl' `geq` vl)
   240          _ -> False

   243    validate :: EscrowParams Datum -> PaymentPubKeyHash -> Action -> ScriptContext -> Bool
   244    validate EscrowParams{escrowDeadline, escrowTargets} contributor action ScriptContext{scriptContextTxInfo} =
   245      case action of
   246        Redeem ->
   247          PlutusTx.traceIfFalse
   248            "escrowDeadline-after"
   249            (escrowDeadline `Interval.after` txInfoValidRange scriptContextTxInfo)
   250            PlutusTx.&& PlutusTx.traceIfFalse
   251              "meetsTarget"
   252              (PlutusTx.all (meetsTarget scriptContextTxInfo) escrowTargets)
   253        Refund ->
   254          PlutusTx.traceIfFalse
   255            "escrowDeadline-before"
   256            ((escrowDeadline PlutusTx.- 1000) `Interval.before` txInfoValidRange scriptContextTxInfo)
   257            PlutusTx.&& PlutusTx.traceIfFalse
   258              "txSignedBy"
   259              (scriptContextTxInfo `txSignedBy` unPaymentPubKeyHash contributor)
